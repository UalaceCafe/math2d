def limit(max)
  msq = squared
  vec = self
  if msq > (max**2)
    vec = vec.div(Math.sqrt(msq))
    vec = vec.mult(max)
  end
  vec
end

def set_magnitude(new_mag)
  mag = magnitude
  mag = mag.zero? ? Float::INFINITY : mag
  Vector2D.new((x * new_mag) / mag, (y * new_mag) / mag)
end
